import sqlite3 # Conecta ao banco de dados (cria se não existir) conn = sqlite3.connect('admin.db') # Cria a tabela 'users' se ela não existir cursor = conn.cursor() cursor.execute(''' CREATE TABLE IF NOT EXISTS users ( id INTEGER PRIMARY KEY AUTOINCREMENT, username TEXT NOT NULL UNIQUE, password TEXT NOT NULL, email TEXT NOT NULL UNIQUE ) ''') conn.commit() # Fecha a conexão conn.close()
